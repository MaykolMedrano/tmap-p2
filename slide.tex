\section{Introducci\'on}

\begin{frame}{Motivaci\'on}
    \begin{itemize}
        \item El paper estudia si el conflicto pol\'itico tiene costos econ\'omicos observables.
        \item La intuici\'on general parece obvia: m\'as violencia deber\'ia da\~nar la econom\'ia.
        \item El problema emp\'irico es m\'as dif\'icil porque la violencia no es un shock ex\'ogeno puro: puede estar ligada a tensiones previas, a la estructura industrial y al ciclo macroecon\'omico.
    \end{itemize}
    \begin{block}{Idea central}
        El desaf\'io central es construir un contrafactual plausible cuando terrorismo y desempe\~no econ\'omico pueden estar end\'ogenamente relacionados.
    \end{block}
\end{frame}

\begin{frame}{Pregunta de investigaci\'on}
    \begin{itemize}
        \item \textquestiondown C\'omo habr\'ia evolucionado la econom\'ia del Pa\'is Vasco en ausencia de terrorismo?
        \item El gran problema es que ese contrafactual no se observa y tampoco es cre\'ible usar al resto de Espa\~na como control bruto.
        \item El paper responde eso principalmente con \emph{synthetic control} y luego lo contrasta con evidencia burs\'atil complementaria.
    \end{itemize}
    \begin{alertblock}{Pregunta operativa}
        Si el conflicto fue costoso, deber\'iamos ver efectos tanto en PIB regional como en valoraci\'on burs\'atil.
    \end{alertblock}
\end{frame}

\begin{frame}{Por qu\'e este paper es interesante}
    \begin{itemize}
        \item El Pa\'is Vasco era una regi\'on relativamente rica antes del conflicto.
        \item Esto descarta la lectura simple de ``violencia como mero s\'intoma de pobreza''.
        \item Adem\'as, este art\'iculo es hist\'orico porque introdujo el m\'etodo de \emph{synthetic control} en la econometr\'ia aplicada.
    \end{itemize}
    \begin{block}{Aporte}
        Este paper es recordado fundacionalmente por mostrar c\'omo construir un contrafactual cre\'ible a nivel regional.
    \end{block}
\end{frame}

\section{Contexto}

\begin{frame}{Contexto hist\'orico}
    \begin{itemize}
        \item ETA se funda en 1959, pero la violencia a gran escala se intensifica desde fines de los 60 y durante los 70.
        \item El conflicto se concentra territorialmente en el Pa\'is Vasco.
        \item Durante d\'ecadas, el terrorismo afect\'o las decisiones de inversi\'on mediante extorsi\'on a empresarios y el deterioro general de las expectativas econ\'omicas.
    \end{itemize}
\end{frame}

\begin{frame}{Algunas cifras del conflicto}
    \centering
    \includegraphics[width=0.62\linewidth]{tablas/tabla_2.png}
    
    \vfill
    
    \small La violencia de ETA estuvo fuertemente concentrada en el Pa\'is Vasco.
\end{frame}

\begin{frame}{Por qu\'e no basta comparar con el resto de Espa\~na}
    \centering
    \includegraphics[width=0.74\linewidth]{tablas/tabla_3.png}
    
    \vfill
    
    \footnotesize
    \begin{itemize}
        \item El Pa\'is Vasco ya era distinto antes del terrorismo.
        \item Compararlo con Espa\~na mezclar\'ia terrorismo con diferencias previas de crecimiento.
    \end{itemize}
    \small Se necesita un contrafactual mejor que ``el promedio de Espa\~na''.
\end{frame}

\begin{frame}{Estrategia de identificaci\'on}
    \begin{itemize}
        \item El paper usa dos dise\~nos complementarios para identificar el costo econ\'omico del conflicto.
        \item El dise\~no principal es el \emph{synthetic control}, que construye un contrafactual regional para medir el efecto de largo plazo sobre el PIB.
        \item La idea es absorber shocks comunes y diferencias previas mejor que en una comparaci\'on simple con el resto de Espa\~na.
        \item La tregua de 1998--1999 aparece como evidencia complementaria, usando un experimento natural en el mercado burs\'atil.
    \end{itemize}
    \begin{block}{Idea}
        Dentro del paper, el control sint\'etico constituye el ejercicio central; el event study aporta evidencia complementaria desde otro margen.
    \end{block}
\end{frame}

\section{Tema central: Synthetic Control}

\begin{frame}{Estrategia emp\'irica 1}
    \begin{itemize}
        \item Los autores construyen un ``Pa\'is Vasco sint\'etico'' usando una combinaci\'on de otras regiones espa\~nolas.
        \item La intuici\'on es simple: buscar regiones que se parezcan al Pa\'is Vasco antes del terrorismo.
        \item Si luego las trayectorias divergen, esa brecha sugiere un costo econ\'omico del conflicto.
    \end{itemize}
\begin{block}{Punto central de la presentaci\'on}
    Esta secci\'on concentra el ejercicio principal del paper: c\'omo se construye el contrafactual y por qu\'e resulta informativo.
\end{block}
\end{frame}

\begin{frame}{C\'omo se arma el control sint\'etico}
    \begin{itemize}
        \item Variables pretratamiento: PIB per c\'apita, inversi\'on, densidad poblacional, estructura sectorial y capital humano.
        \item El procedimiento elige pesos no negativos sobre otras regiones para imitar al Pa\'is Vasco antes del shock.
        \item El ajuste final asigna peso principalmente a Catalu\~na y Madrid.
    \end{itemize}
    \begin{block}{Idea importante}
        Se trata de una regi\'on artificial dise\~nada para parecerse al Pa\'is Vasco antes del terrorismo.
    \end{block}
\end{frame}

\begin{frame}{Control sint\'etico: formulaci\'on m\'inima}
    \begin{equation*}
        W^*=\arg\min_W \left(X_1-X_0W\right)'V\left(X_1-X_0W\right)
    \end{equation*}
    \begin{equation*}
        \text{s.a. } w_j \geq 0 \quad \text{y} \quad \sum_{j=1}^{J} w_j = 1,
        \qquad Y_1^* = Y_0W^*
    \end{equation*}
    \begin{itemize}
        \item $X_1$ resume los predictores pretratamiento del Pa\'is Vasco; $X_0$, de las regiones donantes.
        \item $W$ son pesos que suman 1. La matriz $V$ fija la importancia de cada predictor y emp\'iricamente se elige para minimizar el error de ajuste pretratamiento del PIB.
        \item Con los pesos \'optimos, $Y_1^*=Y_0W^*$ produce la trayectoria contrafactual sin terrorismo.
    \end{itemize}
\end{frame}

\begin{frame}{Por qu\'e esta estrategia es cre\'ible}
    \begin{itemize}
        \item Antes de 1975, el Pa\'is Vasco real y el sint\'etico muestran trayectorias muy parecidas.
        \item Esto sugiere que el contrafactual reproduce adecuadamente la din\'amica previa del ingreso.
        \item La credibilidad no descansa en un supuesto de exogeneidad pura del terrorismo, sino en la capacidad del sint\'etico para absorber shocks comunes y heterogeneidad no observada.
    \end{itemize}
\end{frame}

\begin{frame}{Resultado principal 1: PIB per c\'apita}
    \begin{itemize}
        \item Desde mediados de los 70, el PIB per c\'apita del Pa\'is Vasco cae por debajo del sint\'etico.
        \item La brecha llega a estar cerca de 12\% y en promedio ronda 10\%.
        \item El resultado sugiere un costo persistente del conflicto, no solo transitorio.
    \end{itemize}
    \begin{alertblock}{Hallazgo principal}
        El paper estima una p\'erdida cercana a 10\% en PIB per c\'apita relativo al contrafactual sin terrorismo.
    \end{alertblock}
\end{frame}

\begin{frame}{Placebo con Catalu\~na}
    \begin{itemize}
        \item Para verificar que el m\'etodo no produzca brechas artificiales, aplican la misma idea a Catalu\~na.
        \item Catalu\~na es una regi\'on comparable, pero no sufri\'o una escalada similar de terrorismo.
        \item El resultado placebo no replica la misma brecha prolongada observada en el Pa\'is Vasco.
    \end{itemize}
    \begin{block}{Interpretaci\'on}
        Esto fortalece la idea de que la divergencia del Pa\'is Vasco no es un mero artefacto del m\'etodo.
    \end{block}
\end{frame}

\begin{frame}{Fortalezas y cr\'iticas de esta estrategia}
    \begin{itemize}
        \item Fortaleza: el ajuste pretratamiento y el placebo hacen transparente por qu\'e la comparaci\'on es informativa.
        \item Cr\'itica: el control sint\'etico depende sobre todo de Catalu\~na y Madrid, por lo que la evidencia descansa fuertemente en esa combinaci\'on.
        \item Cr\'itica: la estructura industrial del Pa\'is Vasco no queda replicada perfectamente, lo que genera ruido ante la crisis industrial de los 70.
    \end{itemize}
\end{frame}

\section{Evidencia complementaria: tregua y event study}

\begin{frame}{Transici\'on a la segunda estrategia}
    \begin{itemize}
        \item Hasta aqu\'i, la evidencia central del paper viene del control sint\'etico y del efecto sobre PIB regional.
        \item La segunda parte no reemplaza esa historia: la complementa con reacciones de mercado ante nueva informaci\'on sobre el conflicto.
        \item La idea es usar la tregua de 1998--1999 como validaci\'on adicional, no como el resultado principal del paper.
    \end{itemize}
\end{frame}

\begin{frame}{La tregua de 1998--1999}
    \begin{itemize}
        \item ETA anuncia una tregua en septiembre de 1998.
        \item La credibilidad de la tregua no es inmediata; se construye gradualmente con se\~nales pol\'iticas y contactos con el gobierno.
        \item En noviembre de 1999, ETA anuncia el fin de la tregua.
    \end{itemize}
    \begin{block}{Predicci\'on}
        Si el conflicto tiene un costo econ\'omico, las empresas expuestas al Pa\'is Vasco deber\'ian subir cuando la paz parece cre\'ible y caer cuando colapsa.
    \end{block}
\end{frame}

\begin{frame}{Dise\~no del event study}
    \begin{itemize}
        \item Los autores clasifican firmas en dos grupos: expuestas al Pa\'is Vasco y no expuestas.
        \item Construyen dos portafolios buy-and-hold: 14 acciones ``Basque'' y 59 ``non-Basque''.
        \item Good News cubre 22 sesiones de trading; Bad News, 66 sesiones.
        \item La comparaci\'on se hace con retornos anormales, no solo retornos brutos.
    \end{itemize}
\end{frame}

\begin{frame}{Event study: ecuaci\'on e interpretaci\'on}
    \begin{equation*}
        R_t^j = \alpha^j + \beta_1^j R_t^m + \beta_2^j SMB_t + \beta_3^j HML_t + AR_t^j
    \end{equation*}
    \begin{itemize}
        \item En el paper, $j$ indexa dos portafolios estimados por separado: `Basque` y `non-Basque`.
        \item $R_t^m$, $SMB_t$ y $HML_t$ controlan por movimientos del mercado y factores de riesgo.
        \item En `Table 6`, los autores agregan `Good News` y `Bad News` a esta ecuaci\'on y adem\'as reportan una `difference regression` (`Basque - non-Basque`).
    \end{itemize}
\end{frame}

\begin{frame}{Resultado principal 2}
    \begin{itemize}
        \item Las acciones expuestas al Pa\'is Vasco reaccionan positivamente cuando la tregua gana credibilidad.
        \item Cuando el proceso de paz se deteriora y termina, esas mismas acciones muestran un desempe\~no relativo negativo.
        \item Las firmas no expuestas reaccionan mucho menos.
    \end{itemize}
\end{frame}

\section{Evaluaci\'on general}

\begin{frame}{Aporte del paper}
    \begin{itemize}
        \item Aporta evidencia causal m\'as convincente que una simple correlaci\'on entre conflicto y menor crecimiento.
        \item Su contribuci\'on hist\'orica es introducir formalmente el m\'etodo de \emph{synthetic control} en la literatura econ\'omica.
        \item Un aporte metodol\'ogico clave es volver transparente el contrafactual: qu\'e unidades lo forman, con qu\'e pesos y por qu\'e resulta plausible.
        \item El event study fortalece el argumento, pero el coraz\'on del paper es la construcci\'on del contrafactual sint\'etico.
    \end{itemize}
\end{frame}

\begin{frame}{Limitaciones}
    \begin{itemize}
        \item La objeci\'on m\'as seria es la crisis del petr\'oleo y el declive de la industria pesada en los 70: el Pa\'is Vasco estaba especialmente expuesto a ese shock.
        \item Si el sint\'etico no replica del todo esa composici\'on sectorial, parte de la brecha podr\'ia reflejar cambio estructural y no solo terrorismo.
        \item El control depende mucho de Catalu\~na y Madrid, y la clasificaci\'on de firmas expuestas en el event study descansa en juicio de analistas.
    \end{itemize}
    \begin{block}{Pero}
        La defensa del paper es que combina dos fuentes de evidencia distintas, aunque ninguna elimina por completo esas dudas.
    \end{block}
\end{frame}

\section{Conclusi\'on}

\begin{frame}{Conclusiones}
    \begin{itemize}
        \item El conflicto en el Pa\'is Vasco tuvo costos econ\'omicos relevantes y persistentes.
        \item El resultado central del paper es una brecha de aproximadamente 10\% en PIB per c\'apita respecto del contrafactual sint\'etico.
        \item La evidencia burs\'atil aporta respaldo adicional a esa interpretaci\'on.
    \end{itemize}
    \begin{alertblock}{Cierre}
        En conjunto, el paper muestra que el \emph{synthetic control} ofrece una forma convincente de aproximar un contrafactual no observado.
    \end{alertblock}
\end{frame}
