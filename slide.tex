\begin{frame}{T\'itulo y motivaci\'on}
    \begin{itemize}
        \item Paper: \emph{The Economic Costs of Conflict: A Case Study of the Basque Country}.
        \item Idea central: medir si el terrorismo tiene un costo econ\'omico observable.
        \item El aporte del paper es tratar de ir m\'as all\'a de una correlaci\'on simple entre violencia y menor crecimiento.
    \end{itemize}
\end{frame}

\begin{frame}{Pregunta de investigaci\'on}
    \begin{itemize}
        \item \textquestiondown C\'omo habr\'ia evolucionado la econom\'ia del Pa\'is Vasco sin terrorismo?
        \item El problema clave es que no observamos ese contrafactual.
        \item Por eso el paper necesita construir una comparaci\'on cre\'ible.
    \end{itemize}
\end{frame}

\begin{frame}{Contexto hist\'orico}
    \begin{itemize}
        \item ETA surge en 1959 y la violencia escala desde fines de los 60 y, sobre todo, en los 70.
        \item El conflicto se concentra fuertemente en el Pa\'is Vasco.
        \item Hecho importante: antes del conflicto, el Pa\'is Vasco era una regi\'on relativamente rica dentro de Espa\~na.
    \end{itemize}
    \begin{block}{Dato ilustrativo}
        Casi 70\% de las muertes atribuidas a ETA entre 1968 y 1997 ocurrieron en el Pa\'is Vasco.
    \end{block}
\end{frame}

\begin{frame}{Por qu\'e este caso es interesante}
    \begin{itemize}
        \item No basta comparar al Pa\'is Vasco con el resto de Espa\~na.
        \item El Pa\'is Vasco ya era distinto desde antes en ingreso, industrializaci\'on y capital humano.
        \item Entonces una comparaci\'on ingenua mezclar\'ia el efecto del conflicto con diferencias previas.
    \end{itemize}
\end{frame}

\begin{frame}{Estrategia emp\'irica 1: Synthetic Control}
    \begin{itemize}
        \item Los autores construyen un ``Pa\'is Vasco sint\'etico'' con una combinaci\'on de otras regiones espa\~nolas.
        \item La intuici\'on es usar regiones parecidas antes del terrorismo para aproximar qu\'e habr\'ia pasado sin conflicto.
        \item Luego comparan la trayectoria real del Pa\'is Vasco con la de ese contrafactual sint\'etico.
    \end{itemize}
\end{frame}

\begin{frame}{C\'omo se arma el control sint\'etico}
    \begin{itemize}
        \item Variables pretratamiento: PIB per c\'apita, inversi\'on, densidad, estructura sectorial y capital humano.
        \item El objetivo es replicar bien la senda del Pa\'is Vasco antes del terrorismo.
        \item Resultado importante: el control sint\'etico queda compuesto sobre todo por Catalu\~na y Madrid.
    \end{itemize}
\end{frame}

\begin{frame}{Resultado principal 1}
    \begin{itemize}
        \item Antes de que el terrorismo se vuelva masivo, el Pa\'is Vasco real y el sint\'etico se mueven de forma parecida.
        \item Despu\'es del inicio del terrorismo, ambas trayectorias divergen.
        \item El PIB per c\'apita del Pa\'is Vasco cae alrededor de 10\% respecto del sint\'etico.
    \end{itemize}
    \begin{alertblock}{Hallazgo principal}
        Este es el resultado m\'as fuerte del paper: el conflicto parece tener un costo econ\'omico persistente.
    \end{alertblock}
\end{frame}

\begin{frame}{Evidencia de apoyo}
    \begin{itemize}
        \item Cuando aumenta la intensidad del terrorismo, se ampl\'ia la brecha de PIB.
        \item El efecto aparece con rezagos, consistente con un impacto gradual sobre la actividad econ\'omica.
        \item El placebo con Catalu\~na ayuda a mostrar que la brecha no es solo un artefacto del m\'etodo.
    \end{itemize}
\end{frame}

\begin{frame}{Estrategia emp\'irica 2: tregua de 1998--1999}
    \begin{itemize}
        \item Los autores usan la tregua de ETA como experimento natural.
        \item Si el conflicto da\~na la econom\'ia, las firmas expuestas al Pa\'is Vasco deber\'ian reaccionar positivamente cuando la tregua gana credibilidad.
        \item Para eso comparan acciones de firmas expuestas y no expuestas al Pa\'is Vasco.
    \end{itemize}
\end{frame}

\begin{frame}{Resultado principal 2}
    \begin{itemize}
        \item Las acciones de firmas expuestas al Pa\'is Vasco suben relativamente cuando la tregua se vuelve cre\'ible.
        \item Y caen relativamente cuando la tregua termina.
        \item En cambio, las firmas no expuestas reaccionan mucho menos.
    \end{itemize}
    \begin{table}[htpb]
        \centering

            Per\'iodo & Retorno relativo \\
            \midrule
            Good News & +10.14\% \\
            Bad News & -11.21\% \\
            \bottomrule

    \end{table}
\end{frame}

\begin{frame}{Aporte del paper}
    \begin{itemize}
        \item Aporta evidencia causal m\'as convincente que una correlaci\'on simple entre conflicto y menor crecimiento.
        \item Combina dos estrategias emp\'iricas complementarias: control sint\'etico y event study.
        \item Adem\'as, es un paper cl\'asico porque populariza la intuici\'on del \emph{synthetic control}.
    \end{itemize}
\end{frame}

\begin{frame}{Limitaciones y conclusi\'on}
    \begin{itemize}
        \item Limitaciones: es un caso \'unico, puede haber shocks simult\'aneos, y la clasificaci\'on de firmas expuestas tiene juicio subjetivo.
        \item Aun as\'i, el mensaje general es claro: el conflicto pol\'itico s\'i tuvo costos econ\'omicos relevantes y medibles.
        \item El paper muestra que esos costos aparecen tanto en el PIB regional como en las expectativas del mercado.
    \end{itemize}
